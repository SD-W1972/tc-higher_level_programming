# tests/1-my_list.txt

Testing MyList class
====================

Import the class
>>> MyList = __import__('1-my_list').MyList

Create instance and test print_sorted
>>> my_list = MyList()
>>> my_list.append(1)
>>> my_list.append(4)
>>> my_list.append(2)
>>> my_list.append(3)
>>> my_list.append(5)
>>> print(my_list)
[1, 4, 2, 3, 5]
>>> my_list.print_sorted()
[1, 2, 3, 4, 5]
>>> print(my_list)
[1, 4, 2, 3, 5]

Test with empty list
>>> empty_list = MyList()
>>> empty_list.print_sorted()
[]

Test with single element
>>> single = MyList()
>>> single.append(42)
>>> single.print_sorted()
[42]

Test with already sorted list
>>> sorted_list = MyList()
>>> sorted_list.append(1)
>>> sorted_list.append(2)
>>> sorted_list.append(3)
>>> sorted_list.print_sorted()
[1, 2, 3]

Test with negative numbers
>>> neg_list = MyList()
>>> neg_list.append(-5)
>>> neg_list.append(10)
>>> neg_list.append(-3)
>>> neg_list.append(0)
>>> neg_list.print_sorted()
[-5, -3, 0, 10]
